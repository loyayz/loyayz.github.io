# 添加搜索菜单
menu:
    main:
        - name: 搜索
          url: search/
          weight: 10
params:
    # 添加搜索参数，参考 https://fusejs.io/api/options.html
    fuseOpts:
        isCaseSensitive: false
        includeMatches: true
        shouldSort: true
        location: 0
        distance: 1000
        threshold: 0.4
        minMatchCharLength: 0
        keys: ["title", "permalink", "summary", "content"]
